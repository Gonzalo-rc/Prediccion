  • Test de Lagrange
T.Lag <- function(r,m){
  y = (r - mean(r))^2
  t = length(r)
  a = y [(m +1) : t]
  x = matrix(0, (t - m), m)
  for (i in 1:m){
    x[,i] = y [(m+1-i):(t-i)]
  }
  md = lm(a~x)
  summary(md)
}
